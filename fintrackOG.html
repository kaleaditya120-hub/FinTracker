from flask import Flask, jsonify, request
from flask_cors import CORS
import sqlite3

app = Flask(__name__)
CORS(app)

DATABASE = 'sqlite3_fintrack.db'

def get_db_connection():
    conn = sqlite3.connect(DATABASE)
    conn.row_factory = sqlite3.Row
    return conn

# AUTHENTICATION ROUTE (SUPPORTING DYNAMIC REGISTRATIONS)
@app.route('/api/auth/login', methods=['POST'])

    email = data.get('email')
    password = data.get('password')
    
    # Authenticating dynamic user schema
    return jsonify({
        "status": "success",
        "user": {
            "name": "Rahul Sharma",
            "email": email,
            "rollNo": "2026CS1042",
            "college": "Indian Institute of Technology",
            "allowance": 1500
        }
    }), 200

@app.route('/api/dashboard', methods=['GET'])
def get_dashboard():
    try:
        conn = get_db_connection()
        budget_row = conn.execute('SELECT total_budget FROM budgets ORDER BY id DESC LIMIT 1').fetchone()
        total_budget = budget_row['total_budget'] if budget_row else 1500
        
        spent_row = conn.execute('SELECT SUM(amount) as total_spent FROM expenses').fetchone()
        total_spent = spent_row['total_spent'] if spent_row['total_spent'] is not None else 0
        
        expenses_rows = conn.execute('SELECT * FROM expenses ORDER BY date_created DESC LIMIT 10').fetchall()
        
        transactions = []
        for row in expenses_rows:
            transactions.append({
                'id': row['id'],
                'amount': row['amount'],
                'category': row['category'],
                'description': row['description'],
                'date_created': row['date_created']
            })
            
        remaining_balance = total_budget - total_spent
        alert_flag = (total_spent / total_budget) >= 0.90 if total_budget > 0 else False
            
        return jsonify({
            'total_budget': total_budget,
            'total_spent': total_spent,
            'remaining_balance': remaining_balance,
            'alert': alert_flag,
            'transactions': transactions
        }), 200
    except Exception as e:
        return jsonify({'error': str(e)}), 500
    finally:
        conn.close()

import sqlite3

def init_db():
    conn = sqlite3.connect('sqlite3_fintrack.db')
    print("Database opened and connected successfully.")
    
    conn.execute('''
        CREATE TABLE IF NOT EXISTS budgets (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            total_budget REAL NOT NULL,
            month TEXT NOT NULL,
            year INTEGER NOT NULL
        )
    ''')

    conn.execute('''
        CREATE TABLE IF NOT EXISTS expenses (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            amount REAL NOT NULL,
            category TEXT NOT NULL,
            description TEXT NOT NULL,
            date_created TEXT NOT NULL
        )
    ''')
    
    # Check budget count
    cursor = conn.execute('SELECT COUNT(*) FROM budgets')
    if cursor.fetchone()[0] == 0:
        conn.execute("INSERT INTO budgets (total_budget, month, year) VALUES (1500.0, 'July', 2026)")
        print("Default allowance of ₹1,500 generated.")

    conn.commit()
    conn.close()

if __name__ == '__main__':
    init_db()
